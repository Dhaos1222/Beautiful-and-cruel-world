//从右上角开始搜索
class Solution {
public:
    bool searchMatrix(vector<vector<int>>& matrix, int target) {
        if(matrix.size()==0)return false;
        int y = matrix[0].size()-1;
        int x = 0;
        while(x<matrix.size()&&y>=0)
        {
            if(matrix[x][y]>target)
                --y;
            else if(matrix[x][y]<target)
                ++x;
            else
                return true;
        }
        return false;
        
    }
};
