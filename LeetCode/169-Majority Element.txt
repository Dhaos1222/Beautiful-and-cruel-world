class Solution {
public:
    int majorityElement(vector<int>& nums) {
        // sort(nums.begin(),nums.end());
        // return nums[nums.size()/2];
        int cur;
        int count = 0;
        for(int i = 0;i<nums.size();i++)
        {
            if(count == 0)
                cur = nums[i];
            if(cur==nums[i])
                count++;
            else
                count--;
        }
        return cur;
    }
};
